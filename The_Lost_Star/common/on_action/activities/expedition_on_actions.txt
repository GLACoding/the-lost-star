		exists = scope:activity
		OR = {
			is_ai = no

		}
	}

		chance_of_no_event = {
			if = {
				limit = { this = scope:host }
				value = 0
			}

		}

		200 = expedition.4002 #Gargantuan Warehouse

		#Excavation - move this to a another random pulse

	}
